    Spring Boot ORM Hibernate

    1. Introducción a Hibernate ORM

    En esta sección comprenderás qué es Hibernate y cómo facilita el acceso a datos en Spring Boot.

    Hibernate es el ORM que usa Spring Boot. Permite trabajar con la base de datos mediante entidades Java, evitando escribir SQL directamente. Cada entidad representa una tabla. Spring Data JPA proporciona repositorios automáticos.

    1.1. Instalación

    
<!-- pom.xml -->
<dependency>
    <groupId>org.springframework.boot</groupId>
    <artifactId>spring-boot-starter-data-jpa</artifactId>
</dependency>
<dependency>
    <groupId>com.mysql</groupId>
    <artifactId>mysql-connector-j</artifactId>
    <scope>runtime</scope>
</dependency>

    1.2. Configuración de la BD

    
# application.properties
spring.datasource.url=jdbc:mysql://localhost:3306/hotel_db
spring.datasource.username=root
spring.datasource.password=root
spring.jpa.hibernate.ddl-auto=update
spring.jpa.show-sql=true
spring.jpa.properties.hibernate.format_sql=true

    2. Entidades Hibernate

    Aquí aprenderás a crear entidades y repositorios para representar tablas y operar con registros.

    2.1. Crear una entidad

    
php bin/console make:entity Hotel

    2.2. Entidad básica

    
package com.example.entity;

import jakarta.persistence.*;

@Entity
@Table(name = "hotel")
public class Hotel {

  @Id
  @GeneratedValue(strategy = GenerationType.IDENTITY)
  private Long id;

  @Column(nullable = false)
  private String nombre;

  @Column
  private String direccion;

  // Constructor vacío (requerido por JPA)
  public Hotel() {}

  // Getters y Setters
  public Long getId() { return id; }
  public void setId(Long id) { this.id = id; }

  public String getNombre() { return nombre; }
  public void setNombre(String nombre) { this.nombre = nombre; }

  public String getDireccion() { return direccion; }
  public void setDireccion(String direccion) { this.direccion = direccion; }
}

    Uso en el controlador o servicio:

    
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import java.util.List;

@Service
public class HotelService {

  @Autowired
  private HotelRepository hotelRepository;

  // Insertar
  public Hotel crear() {
    Hotel hotel = new Hotel();
    hotel.setNombre("Gran Hotel");
    hotel.setDireccion("Calle Mayor 3");
    return hotelRepository.save(hotel);
  }

  // Consultar todos
  public List<Hotel> obtenerTodos() {
    return hotelRepository.findAll();
  }

  // Buscar por ID
  public Hotel obtenerPorId(Long id) {
    return hotelRepository.findById(id).orElse(null);
  }

  // Eliminar
  public void eliminar(Long id) {
    hotelRepository.deleteById(id);
  }
}

    4. Relaciones JPA

    Aprenderás a definir relaciones entre entidades para navegar datos conectados.

    4.1. Relación 1:1 (Uno a Uno)

    Una relación 1:1 significa que un registro en una tabla está asociado con exactamente un registro en otra tabla. Ejemplo: Un hotel tiene un único gerente.

    Definición en entidades

    
// src/main/java/com/example/entity/Hotel.java
@Entity
public class Hotel {
  // ... otros campos ...

  @OneToOne(mappedBy = "hotel", cascade = CascadeType.ALL, orphanRemoval = true)
  private Gerente gerente;

  public Gerente getGerente() { return gerente; }
  
  public void setGerente(Gerente gerente) { 
    this.gerente = gerente;
    if (gerente != null) {
      gerente.setHotel(this);
    }
  }
}

    
// src/main/java/com/example/entity/Gerente.java
@Entity
public class Gerente {
  
  @Id
  @GeneratedValue(strategy = GenerationType.IDENTITY)
  private Long id;

  @Column(nullable = false)
  private String nombre;

  private String email;

  @OneToOne
  @JoinColumn(name = "hotel_id", nullable = false)
  private Hotel hotel;

  // Constructor vacío
  public Gerente() {}

  // Getters y Setters
  public Long getId() { return id; }
  public void setId(Long id) { this.id = id; }

  public String getNombre() { return nombre; }
  public void setNombre(String nombre) { this.nombre = nombre; }

  public String getEmail() { return email; }
  public void setEmail(String email) { this.email = email; }

  public Hotel getHotel() { return hotel; }
  public void setHotel(Hotel hotel) { this.hotel = hotel; }
}

    Uso de la relación 1:1

    
// Obtener el gerente de un hotel
Hotel hotel = hotelRepository.findById(1L).orElse(null);
if (hotel != null && hotel.getGerente() != null) {
  System.out.println(hotel.getGerente().getNombre());
}

// Crear un gerente para un hotel
Hotel hotel = hotelRepository.findById(1L).orElse(null);
if (hotel != null) {
  Gerente gerente = new Gerente();
  gerente.setNombre("Juan Pérez");
  gerente.setEmail("juan@hotel.com");
  hotel.setGerente(gerente);
  hotelRepository.save(hotel);
}

    4.2. Relación 1:N (Uno a Muchos)

    Una relación 1:N significa que un registro puede tener múltiples registros relacionados. Ejemplo: Un hotel tiene muchas habitaciones.

    Definición en entidades

    
// src/main/java/com/example/entity/Hotel.java
import java.util.ArrayList;
import java.util.List;

@Entity
public class Hotel {
  // ... otros campos ...

  @OneToMany(mappedBy = "hotel", cascade = CascadeType.ALL, orphanRemoval = true)
  private List<Habitacion> habitaciones = new ArrayList<>();

  public List<Habitacion> getHabitaciones() { 
    return habitaciones; 
  }
  
  public void setHabitaciones(List<Habitacion> habitaciones) { 
    this.habitaciones = habitaciones; 
  }

  // Método helper para agregar habitación
  public void addHabitacion(Habitacion habitacion) {
    habitaciones.add(habitacion);
    habitacion.setHotel(this);
  }

  // Método helper para eliminar habitación
  public void removeHabitacion(Habitacion habitacion) {
    habitaciones.remove(habitacion);
    habitacion.setHotel(null);
  }
}

    4.3. Relación N:M (Muchos a Muchos)

    Una relación N:M significa que múltiples registros pueden estar relacionados con múltiples registros. Ejemplo: Un hotel puede tener muchos servicios, y un servicio puede estar en muchos hoteles. JPA crea automáticamente una tabla intermedia.

    Definición en entidades

    
// src/main/java/com/example/entity/Hotel.java
import java.util.HashSet;
import java.util.Set;

@Entity
public class Hotel {
  // ... otros campos ...

  @ManyToMany(cascade = {CascadeType.PERSIST, CascadeType.MERGE})
  @JoinTable(
    name = "hotel_servicio",
    joinColumns = @JoinColumn(name = "hotel_id"),
    inverseJoinColumns = @JoinColumn(name = "servicio_id")
  )
  private Set<Servicio> servicios = new HashSet<>();

  public Set<Servicio> getServicios() { 
    return servicios; 
  }
  
  public void setServicios(Set<Servicio> servicios) { 
    this.servicios = servicios; 
  }

  // Método helper para agregar servicio
  public void addServicio(Servicio servicio) {
    servicios.add(servicio);
    servicio.getHoteles().add(this);
  }

  // Método helper para eliminar servicio
  public void removeServicio(Servicio servicio) {
    servicios.remove(servicio);
    servicio.getHoteles().remove(this);
  }
}

    
// src/main/java/com/example/entity/Servicio.java
import java.util.HashSet;
import java.util.Set;

@Entity
public class Servicio {
  
  @Id
  @GeneratedValue(strategy = GenerationType.IDENTITY)
  private Long id;

  @Column(nullable = false)
  private String nombre;

  private String descripcion;

  @ManyToMany(mappedBy = "servicios")
  private Set<Hotel> hoteles = new HashSet<>();

  // Constructor vacío
  public Servicio() {}

  // Getters y Setters
  public Long getId() { return id; }
  public void setId(Long id) { this.id = id; }

  public String getNombre() { return nombre; }
  public void setNombre(String nombre) { this.nombre = nombre; }

  public String getDescripcion() { return descripcion; }
  public void setDescripcion(String descripcion) { this.descripcion = descripcion; }

  public Set<Hotel> getHoteles() { return hoteles; }
  public void setHoteles(Set<Hotel> hoteles) { this.hoteles = hoteles; }
}

    Uso de la relación N:M

    
// Obtener servicios de un hotel
Hotel hotel = hotelRepository.findById(1L).orElse(null);
if (hotel != null) {
  for (Servicio servicio : hotel.getServicios()) {
    System.out.println(servicio.getNombre());
  }
}

// Obtener hoteles que tienen un servicio
Servicio servicio = servicioRepository.findById(1L).orElse(null);
if (servicio != null) {
  for (Hotel hotel : servicio.getHoteles()) {
    System.out.println(hotel.getNombre());
  }
}

// Asociar servicios a un hotel
Hotel hotel = hotelRepository.findById(1L).orElse(null);
Servicio servicio = servicioRepository.findById(1L).orElse(null);
if (hotel != null && servicio != null) {
  hotel.addServicio(servicio);
  hotelRepository.save(hotel);
}

// Desasociar servicios
hotel.removeServicio(servicio);
hotelRepository.save(hotel);

// Verificar si tiene un servicio
if (hotel.getServicios().contains(servicio)) {
  System.out.println("El hotel tiene este servicio");
}

// Contar servicios de un hotel
int cantidadServicios = hotel.getServicios().size();

    Relación N:M con atributos extra

    Para agregar campos adicionales en la tabla intermedia (como precio_adicional), debes crear una entidad intermedia explícita: 

    
// src/main/java/com/example/entity/HotelServicio.java
import jakarta.persistence.*;
import java.math.BigDecimal;

@Entity
@Table(name = "hotel_servicio")
public class HotelServicio {
  
  @EmbeddedId
  private HotelServicioId id;

  @ManyToOne
  @MapsId("hotelId")
  @JoinColumn(name = "hotel_id")
  private Hotel hotel;

  @ManyToOne
  @MapsId("servicioId")
  @JoinColumn(name = "servicio_id")
  private Servicio servicio;

  @Column(name = "precio_adicional")
  private BigDecimal precioAdicional;

  // Constructor vacío
  public HotelServicio() {}

  public HotelServicio(Hotel hotel, Servicio servicio, BigDecimal precioAdicional) {
    this.hotel = hotel;
    this.servicio = servicio;
    this.precioAdicional = precioAdicional;
    this.id = new HotelServicioId(hotel.getId(), servicio.getId());
  }

  // Getters y Setters
}

// Clase para la clave compuesta
@Embeddable
public class HotelServicioId implements Serializable {
  
  private Long hotelId;
  private Long servicioId;

  public HotelServicioId() {}

  public HotelServicioId(Long hotelId, Long servicioId) {
    this.hotelId = hotelId;
    this.servicioId = servicioId;
  }

  // Getters, Setters, equals y hashCode
}

    
// Uso con entidad intermedia
Hotel hotel = hotelRepository.findById(1L).orElse(null);
Servicio servicio = servicioRepository.findById(1L).orElse(null);

if (hotel != null && servicio != null) {
  HotelServicio hotelServicio = new HotelServicio(
    hotel, 
    servicio, 
    new BigDecimal("15.00")
  );
  hotelServicioRepository.save(hotelServicio);
}

    5.2. Crear seeder con CommandLineRunner

    
package com.example.seeder;

import com.example.entity.Hotel;
import com.example.repository.HotelRepository;
import net.datafaker.Faker;
import org.springframework.boot.CommandLineRunner;
import org.springframework.stereotype.Component;
import java.util.Locale;

@Component
public class HotelSeeder implements CommandLineRunner {

  private final HotelRepository hotelRepository;

  public HotelSeeder(HotelRepository hotelRepository) {
    this.hotelRepository = hotelRepository;
  }

  @Override
  public void run(String... args) throws Exception {
    if (hotelRepository.count() == 0) {
      Faker faker = new Faker(new Locale("es"));

      for (int i = 0; i < 10; i++) {
        Hotel hotel = new Hotel();
        hotel.setNombre(faker.company().name());
        hotel.setDireccion(faker.address().fullAddress());
        hotelRepository.save(hotel);
      }

      System.out.println("✅ 10 hoteles creados");
    }
  }
}

    6. Repositorios y consultas personalizadas

    Verás cómo crear métodos de búsqueda específicos y consultas personalizadas.

    6.1. Repositorio personalizado

    
// src/main/java/com/example/repository/HotelRepository.java
package com.example.repository;

import com.example.entity.Hotel;
import org.springframework.data.jpa.repository.JpaRepository;
import org.springframework.data.jpa.repository.Query;
import org.springframework.data.repository.query.Param;
import org.springframework.stereotype.Repository;
import java.util.List;

@Repository
public interface HotelRepository extends JpaRepository<Hotel, Long> {

  // Método automático por nombre de campo
  List<Hotel> findByNombreContaining(String nombre);

  // Consulta JPQL personalizada
  @Query("SELECT h FROM Hotel h WHERE h.nombre LIKE %:nombre%")
  List<Hotel> buscarPorNombre(@Param("nombre") String nombre);

  // Consulta SQL nativa
  @Query(value = "SELECT * FROM hotel WHERE nombre LIKE %?1%", nativeQuery = true)
  List<Hotel> buscarPorNombreNativo(String nombre);
}

    6.2. Uso del repositorio

    
List<Hotel> hoteles = hotelRepository.buscarPorNombre("Hotel");
